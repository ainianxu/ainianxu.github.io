文件上传功能开发：增加文件大小格式化过滤器

  • 小技巧：不用算出最终的数值大小，直接写表达式，让程序来计算，这种写法方便维护和修改，比如我写个5242880，看不出来是多少但是写5 * 1024 * 1024，一看就是5M。

  • 再filter.js中设置格式化方法

  • ```js
    /**

    • 格式化文件大小
    • @param value
    • @returns {string}

    */

    let formatFileSize = (value) => {
    value = value || 0;
    let result;
    if (value > 100 * 1024) {
    result = Math.round((value / 1024 / 1024) * 100) / 100 + “MB”;
    } else {
    result = Math.round((value / 1024) * 100) / 100 + “KB”;
    }
    return result;
    };
    //将其抛出
    export default {
    formatFileSize
    }

  mounted: function() {

     * 打开内容编辑框

      $("#content").summernote({
        focus: true,
        height: 300
      });

      $("#content").summernote('code', '');

      // 加载内容文件列表

        Loading.hide();

        if (resp.success) {
          if (resp.content) {
            $("#content").summernote('code', resp.content.content);

          // 定时自动保存
